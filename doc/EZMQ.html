def self.create_linked_pair(**options)
  options[:context] ||= EZMQ::Context.new
  options[:protocol] ||= 'inproc'
  options[:address] ||= options[:context].context.address
  %i(bind connect).map do |mode|
    EZMQ::Pair.new mode, options
  end
end
