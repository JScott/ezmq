def listen
  loop do
    if block_given?
      yield(*receive)
    else
      message, topic = receive
      puts "#{ topic } #{ message }"
    end
  end
end

def receive(**options)
  message = ''
  @socket.recv_string message

  message = message.match(/^(?<topic>[^\ ]*)\ (?<body>.*)/)

  decoded = (options[:decode] || @decode).call message['body']
  if block_given?
    yield decoded, message['topic']
  else
    [decoded, message['topic']]
  end
end
